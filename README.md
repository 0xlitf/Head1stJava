```
    <proxy>
      <id>optional</id>
      <active>true</active>
      <protocol>http</protocol>
      <username></username>
      <password></password>
      <host>127.0.0.1</host>
      <port>20809</port>
      <nonProxyHosts></nonProxyHosts>
    </proxy>
  </proxies>
```
